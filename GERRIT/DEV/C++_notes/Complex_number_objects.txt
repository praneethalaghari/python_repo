#include <iostream>

using namespace std;

class Complex
{
  public:
    int real;
    int img;

  void print()
  {
      cout << real << "+" << img << "j" << endl; 
  }
  
   Complex operator +(const Complex& x1)
  {
      Complex temp;
      temp.real = real + x1.real;
      temp.img = img + x1.img;
      return temp;
  }
  
   Complex operator -(const Complex& x1)
  {
      Complex temp;
      temp.real = real - x1.real;
      temp.img = img - x1.img;
      return temp;
  }
 
    
};
