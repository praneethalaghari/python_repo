#include <bits/stdc++.h>

using namespace std;

int main()
{
    int arr[] = {-2,-1,-3,-4,-1,-2,-1,-5,-4};
    
    int n = sizeof(arr)/sizeof(arr[0]);
    
    int curr_sum = 0;
    int max_sum = arr[0];
    
    for(int i = 0; i < n; ++i)
    {
     curr_sum = curr_sum + arr[i];
     
     if(curr_sum < 0)               // If during cumulative addition if curr_sum is negative then make it 0
     {
         curr_sum = 0;
     }
     
     if(curr_sum > max_sum)
     {
         max_sum = curr_sum;
     }
    
    }
    
    cout << max_sum << endl;
    
    return 0;
}
