#include <iostream>
#include <map>
#include <cstring>
#include <cmath>

using namespace std;

int main()
{
     int a[] = {1,4,7,3,5,6}; // Sum = 8
     int sum = 10;
     
     int n = sizeof(a)/sizeof(a[0]);
     
     map<int,int> m1;
     
     for(int i=0; i<n; ++i)
     {
         m1.insert(pair<int,int>(a[i],sum-a[i]));
     }
     
     int pairs = 0;
     map<int, int>::iterator it;
     for(it=m1.begin();it!=m1.end();++it)
     {
       map<int,int>::iterator it1 = m1.begin();
       it1 = m1.find(it->second);
       if(it1!=m1.end())
       {
           pairs++;
       }
     }
     
     cout << ceil(float(pairs)/2) << endl;  // Divided by 2 to avoid duplicate pairs i.e 1,7 and 7,1
     
    return 0;
}
