#include <iostream>
#include <string>
#include <typeinfo>
#include <cxxabi.h>

using namespace std;

int main()
{
    string name1 = "Praneeth";
    string name2 = "Srinivas";
    string name3 = "Alaghari";
    
    cout << abi::__cxa_demangle(typeid(name1).name(),0,0,0) <<endl;
    cout << abi::__cxa_demangle(typeid(name1.c_str()).name(),0,0,0) << endl;
    
    const char* cc_name = name1.c_str();
    
    string sname = string(cc_name);
    cout << sname << endl;
    
    cout << *(name1.c_str()+1) << endl;
    return 0;
}
